\def\title[#1]{
    \newpage
    \ifhmode\par\fi
    \removelastskip
    \vskip 1ex\goodbreak
    \refstepcounter{chapter}%
    \noindent
    \begingroup
    \leavevmode\Large\bfseries\raggedright
    \rlap{\thechapter}%
    \hspace{\titleindent} \huge Title \arabic{chapter}. %
    #1
    \par
    \endgroup
    \addcontentsline{toc}{chapter}{\protect\numberline{\thechapter}#1}
    %\vskip 0ex \nobreak
    %\removelastskip
    \leftskip=12mm%\relax%
}

\def\article[#1]{
    \ifhmode\par\fi
    \removelastskip
    \vskip 1ex\goodbreak
    \refstepcounter{section}%
    \noindent
    \begingroup
    \leavevmode\Large\bfseries\raggedright
    \rlap{\thesection}%
    \hspace{\sectionindent} Article \Roman{section}. \hspace{0.1mm} %
    #1
    \par
    \endgroup
    \addcontentsline{toc}{section}{\protect\numberline{\thesection}#1}
    %\vskip 0ex \nobreak
    %\removelastskip
    \leftskip=12mm%\relax%
}
